### –ü–æ—á–µ–º—É –ø—Ä–æ—Å—Ç–æ–µ —Å–ª–æ–∂–µ–Ω–∏–µ –Ω–µ —Ä–∞–±–æ—Ç–∞–µ—Ç

–ï—Å–ª–∏ –±–∞–Ω–∫ –ø—Ä–æ—Å—Ç–æ —Å–ª–æ–∂–∏—Ç —ç—Ç–∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ –∫–ª–∞—Å—Å–∏—á–µ—Å–∫–∏–º —Å–ø–æ—Å–æ–±–æ–º, –ø–æ–ª—É—á–∏—Ç—Å—è: P(–ª—é–±–æ–π —Ä–∏—Å–∫) = 1 - (1-0.12)√ó(1-0.08)√ó(1-0.18)√ó(1-0.15) ‚âà 45%. –≠—Ç–æ –æ–∑–Ω–∞—á–∞–µ—Ç –æ—Ç–∫–∞–∑ –ø—Ä–∞–∫—Ç–∏—á–µ—Å–∫–∏ –∫–∞–∂–¥–æ–º—É –≤—Ç–æ—Ä–æ–º—É –∫–ª–∏–µ–Ω—Ç—É!

–ù–æ –∞–Ω–∞–ª–∏—Ç–∏–∫–∏ –±–∞–Ω–∫–∞ –ø–æ–Ω–∏–º–∞—é—Ç: –º–Ω–æ–≥–∏–µ –∏–∑ —ç—Ç–∏—Ö —Å–∏–≥–Ω–∞–ª–æ–≤ —Å–≤—è–∑–∞–Ω—ã. –ê–Ω—Ç–∏-—Ñ—Ä–æ–¥ –∏ —Å–∫–æ—Ä–∏–Ω–≥–æ–≤–∞—è –º–æ–¥–µ–ª—å —á–∞—Å—Ç–æ —Ä–µ–∞–≥–∏—Ä—É—é—Ç –Ω–∞ –æ–¥–Ω–∏ –∏ —Ç–µ –∂–µ –ø–æ–¥–æ–∑—Ä–∏—Ç–µ–ª—å–Ω—ã–µ –ø–∞—Ç—Ç–µ—Ä–Ω—ã –ø–æ–≤–µ–¥–µ–Ω–∏—è. –ê —Ä—É—á–Ω–∞—è –ø—Ä–æ–≤–µ—Ä–∫–∞ –æ–±—ã—á–Ω–æ **—Å–Ω–∏–∂–∞–µ—Ç** –ø–æ–¥–æ–∑—Ä–µ–Ω–∏—è, –Ω–∞–π–¥–µ–Ω–Ω—ã–µ –∞–≤—Ç–æ–º–∞—Ç–∏–∫–æ–π, –ø–æ—Ç–æ–º—É —á—Ç–æ –∂–∏–≤–æ–π —á–µ–ª–æ–≤–µ–∫ –º–æ–∂–µ—Ç –æ–±—ä—è—Å–Ω–∏—Ç—å —Å—Ç—Ä–∞–Ω–Ω–æ—Å—Ç–∏.

### –ö–∞–∫ –º—ã —ç—Ç–æ –º–æ–¥–µ–ª–∏—Ä—É–µ–º —É–≥–ª–∞–º–∏

**–î—É–±–ª–∏ (—É–≥–æ–ª 0¬∞):** –ê–Ω—Ç–∏-—Ñ—Ä–æ–¥ (p=0.12) –∏ —Å–∫–æ—Ä–∏–Ω–≥ (p=0.18) —á–∞—Å—Ç–æ —Å—Ä–∞–±–∞—Ç—ã–≤–∞—é—Ç –Ω–∞ –æ–¥–Ω–æ –∏ —Ç–æ –∂–µ ‚Äî –Ω–æ—á–Ω—ã–µ –∑–∞—è–≤–∫–∏, –Ω–µ—Å—Ç–∞–±–∏–ª—å–Ω—ã–µ –¥–æ—Ö–æ–¥—ã, —Å—Ç—Ä–∞–Ω–Ω—ã–µ IP. –°—Ç–∞–≤–∏–º –∏–º —É–≥–æ–ª 0¬∞, —á—Ç–æ–±—ã –Ω–µ —É–¥–≤–∞–∏–≤–∞—Ç—å –æ–¥–∏–Ω–∞–∫–æ–≤—ã–µ –ø–æ–¥–æ–∑—Ä–µ–Ω–∏—è.

**–ö–æ–º–ø–µ–Ω—Å–∞—Ü–∏—è (—É–≥–æ–ª 180¬∞):** –†—É—á–Ω–∞—è –ø—Ä–æ–≤–µ—Ä–∫–∞ (p=0.15) –æ–±—ã—á–Ω–æ —Ä–∞–±–æ—Ç–∞–µ—Ç ¬´–ø—Ä–æ—Ç–∏–≤¬ª –∞–≤—Ç–æ–º–∞—Ç–∏–∫–∏ ‚Äî –µ—Å–ª–∏ —Ä–æ–±–æ—Ç –Ω–∞—à—ë–ª –ø–æ–¥–æ–∑—Ä–µ–Ω–∏—è, —á–µ–ª–æ–≤–µ–∫ –º–æ–∂–µ—Ç –∏—Ö –æ–ø—Ä–æ–≤–µ—Ä–≥–Ω—É—Ç—å. –°—Ç–∞–≤–∏–º —É–≥–æ–ª 180¬∞.

**–ù–µ–∑–∞–≤–∏—Å–∏–º–æ—Å—Ç—å (–ø—Ä–æ–º–µ–∂—É—Ç–æ—á–Ω—ã–π —É–≥–æ–ª):** –ö—Ä–µ–¥–∏—Ç–Ω–æ–µ –±—é—Ä–æ (p=0.08) —Ä–∞–±–æ—Ç–∞–µ—Ç —Å –∏—Å—Ç–æ—Ä–∏—á–µ—Å–∫–∏–º–∏ –¥–∞–Ω–Ω—ã–º–∏, –∫–æ—Ç–æ—Ä—ã–µ –Ω–µ –∑–∞–≤–∏—Å—è—Ç –æ—Ç —Ç–µ–∫—É—â–µ–≥–æ –ø–æ–≤–µ–¥–µ–Ω–∏—è. –°—Ç–∞–≤–∏–º —É–≥–æ–ª œÄ/3 (60¬∞).

### –†–µ—Ü–µ–ø—Ç (–∫–æ–ø–∏—Ä—É–π—Ç–µ —Ü–µ–ª–∏–∫–æ–º):

```python
# –ë–õ–û–ö 1: –ë–∞–∑–æ–≤–∞—è –æ—Ü–µ–Ω–∫–∞ (–ø—Ä–æ—Å—Ç–æ–µ —Å–ª–æ–∂–µ–Ω–∏–µ)
risk_sources = [0.12, 0.08, 0.18, 0.15]  # –≤–∞—à–∏ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ —Ä–∏—Å–∫–æ–≤
print("–ò—Å—Ö–æ–¥–Ω—ã–µ —Ä–∏—Å–∫–∏:", risk_sources)

# –ü—Ä–æ—Å—Ç–æ–µ —Å–ª–æ–∂–µ–Ω–∏–µ –Ω–∞ —É—Ä–æ–≤–Ω–µ 2
base_risk = ProbabilisticBit(risk_sources[0])
for p in risk_sources[1:]:
    base_risk = base_risk | ProbabilisticBit(p)
print("–ë–∞–∑–æ–≤—ã–π —Ä–∏—Å–∫ (–ø—Ä–æ—Å—Ç–æ–µ —Å–ª–æ–∂–µ–Ω–∏–µ):", round(base_risk.probability, 4))

# –ë–õ–û–ö 2: –£–º–Ω–æ–µ –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ —Å —É—á—ë—Ç–æ–º –¥—É–±–ª–µ–π/–∫–æ–Ω—Ñ–ª–∏–∫—Ç–æ–≤
# –£–≥–ª—ã –ø–æ –ª–æ–≥–∏–∫–µ –Ω–∞—à–µ–≥–æ –±–∞–Ω–∫–æ–≤—Å–∫–æ–≥–æ –ø—Ä–∏–º–µ—Ä–∞:
# –ê–Ω—Ç–∏-—Ñ—Ä–æ–¥ (0.12) –∏ —Å–∫–æ—Ä–∏–Ω–≥ (0.18) = –¥—É–±–ª–∏ = —É–≥–æ–ª 0¬∞
# –ö—Ä–µ–¥–∏—Ç–Ω–æ–µ –±—é—Ä–æ (0.08) = –Ω–µ–∑–∞–≤–∏—Å–∏–º–æ = —É–≥–æ–ª œÄ/3
# –†—É—á–Ω–∞—è –ø—Ä–æ–≤–µ—Ä–∫–∞ (0.15) = –∫–æ–º–ø–µ–Ω—Å–∞—Ü–∏—è = —É–≥–æ–ª œÄ
angles = [0.0, np.pi/3, 0.0, np.pi]  # [–∞–Ω—Ç–∏-—Ñ—Ä–æ–¥, –∫—Ä–µ–¥.–±—é—Ä–æ, —Å–∫–æ—Ä–∏–Ω–≥, —Ä—É—á–Ω–∞—è]
states = [PhaseState(p, angle) for p, angle in zip(risk_sources, angles)]

# –í—ã–±–µ—Ä–∏—Ç–µ —Ä–µ–∂–∏–º —Å–∫–ª–µ–π–∫–∏ (–ø–æ–ø—Ä–æ–±—É–π—Ç–µ —Ä–∞–∑–Ω—ã–µ)
set_phase_or_mode('weight')  # —Ä–æ–±–∞—Å—Ç–Ω—ã–π —Ä–µ–∂–∏–º –¥–ª—è –ø—Ä–æ–¥–∞–∫—à–µ–Ω–∞
# set_phase_or_mode('norm')  # –∫–æ–Ω—Å–µ—Ä–≤–∞—Ç–∏–≤–Ω—ã–π –¥–ª—è —Ä–µ–≥—É–ª—è—Ç–æ—Ä–∏–∫–∏
# set_phase_or_mode('quant') # –∞–≥—Ä–µ—Å—Å–∏–≤–Ω—ã–π –¥–ª—è –∏—Å—Å–ª–µ–¥–æ–≤–∞–Ω–∏–π

smart_risk = states[0]
for state in states[1:]:
    smart_risk = smart_risk | state

print("–£–º–Ω—ã–π —Ä–∏—Å–∫ (—Å —É—á—ë—Ç–æ–º –¥—É–±–ª–µ–π):", round(smart_risk.probability, 4))

# –ë–õ–û–ö 3: –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ —É—Å—Ç–æ–π—á–∏–≤–æ—Å—Ç—å –∫ —à—É–º–∞–º
noisy_risk = smart_risk
noisy_risk = dephase(noisy_risk, sigma_phi=0.1)   # —Ñ–∞–∑–æ–≤—ã–π —à—É–º
noisy_risk = amp_damp(noisy_risk, alpha=0.2)      # –∑–∞—Ç—É—Ö–∞–Ω–∏–µ
print("–†–∏—Å–∫ –ø–æ–¥ —à—É–º–æ–º:", round(noisy_risk.probability, 4))

# –ë–õ–û–ö 4: –î–∏–∞–≥–Ω–æ—Å—Ç–∏–∫–∞ –∏–Ω—Ñ–æ—Ä–º–∞—Ç–∏–≤–Ω–æ—Å—Ç–∏
from probstates.entropy import calculate_entropy
h_base = calculate_entropy(base_risk)
h_smart = calculate_entropy(smart_risk)
print("–≠–Ω—Ç—Ä–æ–ø–∏—è –±–∞–∑–æ–≤–æ–≥–æ —Ä–∏—Å–∫–∞:", round(h_base, 3))
print("–≠–Ω—Ç—Ä–æ–ø–∏—è —É–º–Ω–æ–≥–æ —Ä–∏—Å–∫–∞:", round(h_smart, 3))
```

- –î–∞—Ç—á–∏–∫ CO –æ—á–µ–Ω—å —Ç–æ—á–µ–Ω, –Ω–æ –¥–∞–ª–µ–∫–æ —Ä–∞—Å–ø–æ–ª–æ–∂–µ–Ω

### –ö–∞–∫ –º—ã —ç—Ç–æ –º–æ–¥–µ–ª–∏—Ä—É–µ–º

- –¢–µ–ø–ª–æ–≤–æ–π –¥–∞—Ç—á–∏–∫ (0.65): —É–≥–æ–ª 30¬∞ (œÄ/6) ‚Äî —Å–∏—Å—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–∏–π —Å–¥–≤–∏–≥ –∏–∑-–∑–∞ –º–µ–¥–ª–µ–Ω–Ω–æ–π —Ä–µ–∞–∫—Ü–∏–∏

### –†–µ—Ü–µ–ø—Ç (–∫–æ–ø–∏—Ä—É–π—Ç–µ —Ü–µ–ª–∏–∫–æ–º):

```python
sensor_probs = [0.7, 0.65, 0.8]           # –Ω–∞–¥—ë–∂–Ω–æ—Å—Ç—å –∫–∞–∂–¥–æ–≥–æ —Ç–∏–ø–∞
sensor_angles = [0.0, np.pi/6, np.pi/4]   # –∫–∞–ª–∏–±—Ä–æ–≤–æ—á–Ω—ã–µ —Å–¥–≤–∏–≥–∏ –æ—Ç –∑–∞–≤–æ–¥–∞
print("–î–∞—Ç—á–∏–∫–∏:", [(p, f"—É–≥–æ–ª={round(np.degrees(a))}¬∞") for p, a in zip(sensor_probs, sensor_angles)])

sensors = []

    s = PhaseState(p, angle)
    s = amp_damp(s, alpha=0.15)             # –∑–∞—Ç—É—Ö–∞–Ω–∏–µ —Å–∏–≥–Ω–∞–ª–∞

    sensors.append(s)

# –ë–õ–û–ö 3: –°–ª–∏—è–Ω–∏–µ –¥–∞—Ç—á–∏–∫–æ–≤
set_phase_or_mode('weight')  # —Ä–æ–±–∞—Å—Ç–Ω—ã–π —Ä–µ–∂–∏–º

for sensor in sensors[1:]:
    fused = fused | sensor

# –ë–õ–û–ö 4: –ï—Å–ª–∏ –º–Ω–æ–≥–æ –≥–∏–ø–æ—Ç–µ–∑ ‚Äî –∏—Å–ø–æ–ª—å–∑—É–π—Ç–µ —Ä–µ–≥–∏—Å—Ç—Ä

reg = PhaseRegister.uniform(3)  # 2^3 = 8 –≥–∏–ø–æ—Ç–µ–∑

best_hypothesis = reg.argmax_probability()
print("–õ—É—á—à–∞—è –≥–∏–ø–æ—Ç–µ–∑–∞ (–∏–Ω–¥–µ–∫—Å):", best_hypothesis)
print("–ï—ë –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å:", round(reg.measure_probability(best_hypothesis), 4))
```

- –ú–æ–±–∏–ª—å–Ω—ã–µ (12%‚Üí14.5%) –∏ –¥–µ—Å–∫—Ç–æ–ø–Ω—ã–µ (11%‚Üí13%) ‚Äî —Å–æ–≥–ª–∞—Å–æ–≤–∞–Ω—ã, –æ–±–∞ –ø–æ–∫–∞–∑—ã–≤–∞—é—Ç +2% –≤ –ø–æ–ª—å–∑—É B

```python
h_a = shannon_entropy(pA)

print(f"–ù–µ–æ–ø—Ä–µ–¥–µ–ª—ë–Ω–Ω–æ—Å—Ç—å A: {round(h_a, 3)}, B: {round(h_b, 3)}")
print(f"–û—Ç–ª–∏—á–∏–µ A‚ÜíB: {round(kl_ab, 3)}, B‚ÜíA: {round(kl_ba, 3)}")

if max(kl_ab, kl_ba) < 0.01:
    print("‚ö†Ô∏è  –†–∞–∑–ª–∏—á–∏—è —Å–ª–∞–±—ã–µ, –Ω—É–∂–Ω–æ –±–æ–ª—å—à–µ –¥–∞–Ω–Ω—ã—Ö")

# [–º–æ–±–∏–ª—å–Ω—ã–µ, –¥–µ—Å–∫—Ç–æ–ø–Ω—ã–µ, –Ω–æ–≤—ã–µ, –ø–æ—Å—Ç–æ—è–Ω–Ω—ã–µ] 
channels_A = [0.12, 0.11, 0.125, 0.14]  # –∫–æ–Ω–≤–µ—Ä—Å–∏–∏ A –ø–æ —Å–µ–≥–º–µ–Ω—Ç–∞–º
```
